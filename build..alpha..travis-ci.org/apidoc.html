function get-urls (str, opts)

  • description and source-code
    (str, opts) => {
	const urls = str.match(urlRegex());
	const ret = urls ? urls.map(url => normalizeUrl(url.trim().replace(/\.+$/, ''), opts)) : [];
	return new Set(ret);
}
